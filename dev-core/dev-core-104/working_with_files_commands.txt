


touch - создать файл
echo - записать в файл
ls - посмотреть файлы в директории 
cat - посмотреть содержимое файла
head - посмотреть первые 10 строк файла (-n 2 (первые 2))
tail - посмотреть последние 10 строк файла (-n 1 (последняя 1))
mkdir - создать директорию
mv - перенести или переименовать файл
cp - скопировать файл 
rm - удалить файл
